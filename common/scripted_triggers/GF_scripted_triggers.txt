# scope = country

@high_zion_favor_threshold	 = 100
@medium_zion_favor_threshold = 50				# currently no use...
@low_zion_favor_threshold	 = 15

GF_has_zion_favor_high = {
	is_variable_set = v_GF_zion_favor
	check_variable = {
		which = v_GF_zion_favor
		value >= @high_zion_favor_threshold
	}
}

GF_has_zion_favor_medium = {
	is_variable_set = v_GF_zion_favor
	AND = {
		check_variable = {
			which = v_GF_zion_favor
			value < @high_zion_favor_threshold
		}
		check_variable = {
			which = v_GF_zion_favor
			value > @low_zion_favor_threshold
		}
	}
}

GF_has_zion_favor_low = {
	is_variable_set = v_GF_zion_favor
	check_variable = {
		which = v_GF_zion_favor
		value <= @low_zion_favor_threshold
	}
}
