GF_doll_reprogram = {
	length = 0
	icon = "GFX_edict_type_time"
	resources = {
		category = edicts
		cost = { unity = 1000 }
		cost = {
			energy = 1000
			multiplier = value:GF_count_robotic_leader_with_vanilla_negative_traits
		}
	}

	allow = {
		custom_tooltip = {
			fail_text = GF_require_any_doll_leader_has_vanilla_negative_traits
			count_owned_leader = {
				limit = {
					species = { is_robotic = yes }
					GF_leader_has_vanilla_negative_traits = yes
				}
				count > 0
			}
		}
		NOT = { has_modifier = GF_edict_doll_reprogram_cooldown }
	}

	potential = {
		has_authority = GF_auth_commonwealth
		has_technology = GF_tech_mind_upgrade_2
	}
	effect = {
		custom_tooltip = GF_doll_reprogram_effect
		hidden_effect = {
			add_modifier = {
				modifier = GF_edict_doll_reprogram_cooldown
			}
			country_event = { id = GF_doll_leader.100 }
		}
	}

	ai_weight = {
		weight = 0
		modifier = {
			add = @EdictHighPrio
			count_owned_leader = {
				limit = { GF_leader_has_vanilla_negative_traits = yes }
				count > 0
			}
		}
	}
}
