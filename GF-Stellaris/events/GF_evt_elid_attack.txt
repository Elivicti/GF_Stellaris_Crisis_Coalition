### 污染区存在时，刷新ELID进攻事件

# 通用ELID进攻刷新事件
planet_event = {
	id = GF_elid.500
	hide_window = yes

	is_triggered_only = yes
	trigger = {
		GF_elid_threat_exists = yes
	}

	immediate = {
		if = {
			limit = {
				has_ground_combat = no
				OR = {
					is_occupied_flag = no
					controller = {
						NOT = { is_country_type = GF_elid_country }
					}
				}
			}
			fire_on_action = {
				on_action = on_GF_elid_invasion_spawned
				scopes = {
					from = event_target:GF_ogas_owner
				}
			}
			# 正常刷出后，立即启动下一次刷新事件
			planet_event = {
				id = GF_elid.500
				days = @elid_attack_interval
				random = @elid_attack_interval_random
			}
		}
		else = { # 若星球已被占领/正在进行地面作战，45到90天后再次检查
			planet_event = {
				id = GF_elid.500
				days = 45
				random = 45
			}
		}
	}
}

			text = NEW_LINE

			}
			success_text = {
				text = GF_elid.1010.desc.no_further
				GF_planet_has_contaminated_zone = no
			}
		}
	}

	picture = GFX_evt_crisis_defeated
	show_sound = event_celebration
	location = this

	is_triggered_only = yes
	trigger = {
		GF_elid_threat_exists = yes
		GF_check_elid_combat_type = { TYPE = regular }
		owner = {
			GF_is_main_country = yes
		}
	}

	option = {
		name = GF_elid.1010.a
		owner = {
			add_resource = {
				minerals = 200
				mult = prev.v_GF_elid_threat_level
			}
		}
		locked_random_list = {
			# 威胁度大于2时，一定概率减少威胁度
			5  = {
				modifier = {
					add = 15
					GF_planet_has_contaminated_zone = no
				}
				modifier = {
					factor = 0
					check_variable = {
						which = v_GF_elid_threat_level
						value > @GF_elid_threat_low_threshold
					}
				}
				GF_decrease_elid_threat = { VALUE = 1 }
			}
			95 = {
				modifier = {
					subtract = 15
					GF_planet_has_contaminated_zone = no
				}
			}
		}
		add_modifier = {
			modifier = GF_pm_victory_in_defending_elid
			days = 90
		}
	}
}

		}

			set_variable = {
				which = tmp_devastation
				value = 0
			}
			# 根据威胁度添加随机破坏度
			locked_random_list = {
				1 = {
					modifier = {
						add = 10
						GF_planet_has_contaminated_zone = no
					}
					modifier = {
						factor = 2
						has_modifier = GF_pm_elid_threat_level_low

						factor = 1.2

					}
				}
				3 = {
					modifier = {
						factor = 2
						has_modifier = GF_pm_elid_threat_level_low
					}
					modifier = {
						factor = 1.5
						has_modifier = GF_pm_elid_threat_level_mid
					}
					set_variable_to_random_value = {
						which = tmp_devastation
						min = 1
						max = 6
					}
				}
				2 = {
					modifier = {
						factor = 0.1
						has_modifier = GF_pm_elid_threat_level_low
					}
					modifier = {
						factor = 3
						has_modifier = GF_pm_elid_threat_level_mid
					}
					modifier = {
						factor = 1.5
						has_modifier = GF_pm_elid_threat_level_high
					}
					set_variable_to_random_value = {
						which = tmp_devastation
						min = 3
						max = 9
					}
				}
				2 = {
					modifier = {
						factor = 0
						has_modifier = GF_pm_elid_threat_level_low
					}
					modifier = {
						factor = 2
						has_modifier = GF_pm_elid_threat_level_mid
					}
					modifier = {
						factor = 1.2
						has_modifier = GF_pm_elid_threat_level_high
					}
					set_variable_to_random_value = {
						which = tmp_devastation
						min = 7
						max = 13
					}
				}
				1 = {
					modifier = {
						factor = 0
						has_modifier = GF_pm_elid_threat_level_low
					}
					modifier = {
						factor = 0.5
						has_modifier = GF_pm_elid_threat_level_mid
					}
					modifier = {
						factor = 2
						has_modifier = GF_pm_elid_threat_level_high
					}
					set_variable_to_random_value = {
						which = tmp_devastation
						min = 10
						max = 20
					}
				}
			}
		}
		if = {
			limit = {
				check_variable = {
					which = tmp_devastation
					value > 0
				}
			}
			add_planet_devastation = tmp_devastation
		}
	}

	after = {
		hidden_effect = {
			clear_variable = tmp_devastation
			GF_remove_elid_invasion_army = yes
		}
		add_deposit = d_GF_yellow_zone_extra
	}
}
