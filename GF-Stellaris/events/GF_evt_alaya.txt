		}
	}

	option = {
		name = GF_alaya.004.a

		end_event_chain = "GF_collect_relic_device_event_chain"

		add_modifier = {
			modifier = "GF_productive_relic_device"
			days = -1
		}

		hidden_effect = {
			country_event = {
				id = GF_alaya.100
				days = 1440
				random = 1440
			}
		}
	}
}

	}
