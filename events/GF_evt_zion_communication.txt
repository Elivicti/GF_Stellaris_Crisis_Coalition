country_event = {
	id = GF_zion.0

	is_triggered_only = yes
	hide_window = yes

	trigger = {
		GF_default_zion_comm_trigger = yes
	}

	immediate = {
		if = {
			limit = { NOT = { has_country_flag = flag_GF_first_zion_comm_completed } }
			country_event = { 
				id = GF_zion.10000
			}
		}
		else = {
			country_event = {
				id = GF_zion.20000
			}
		}
	}
}
